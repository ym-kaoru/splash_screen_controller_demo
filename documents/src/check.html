state_machine('Abc', name:'sm') {
    def_public_event :onCreate => [arg('saved_instance_state', type:'Bundle')]
    def_public_event :notifyTimeout
    def_public_event :click

    state(:ENTRY_POINT) {
        onCreate :SPLASH_SCREEN
    }

    state(:SPLASH_SCREEN, entry:'self.parent_context.showSplashScreen()', exit:'self.parent_context.hideSplashScreen()') {
        notifyTimeout :DOOR
    }

    state(:DOOR, entry:'self.parent_context.replaceDoorFragment()') {
        click :MAIN
    }

    state(:MAIN, entry:'self.parent_context.replaceMainFragment()') {
    }
}
